-- ============================================
-- 1. 유저 관련 테이블
-- ============================================

-- 1) USERS

CREATE TABLE users (
    user_id BIGSERIAL PRIMARY KEY,
    username VARCHAR(255) NOT NULL,
    password_hash VARCHAR(255),
    nickname VARCHAR(50) NOT NULL,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP,
    deleted_at TIMESTAMP
);

-- username 활성 사용자만 UNIQUE
CREATE UNIQUE INDEX idx_users_username_active
ON users(username)
WHERE deleted_at IS NULL;

-- 2) USER_PROFILE  (1:1)

CREATE TABLE user_profile (
    user_id BIGINT PRIMARY KEY,
    profile_image_url VARCHAR(500) NOT NULL,
    reminder_enabled BOOLEAN NOT NULL DEFAULT FALSE,
    reminder_time TIME,
    CONSTRAINT fk_user_profile_user
        FOREIGN KEY (user_id)
        REFERENCES users(user_id)
);


-- 3) SOCIAL_ACCOUNT (OAuth 로그인 정보 저장)

CREATE TABLE social_account (
    social_id BIGSERIAL PRIMARY KEY,
    user_id BIGINT NOT NULL,
    provider VARCHAR(20) NOT NULL,
    provider_user_id VARCHAR(255) NOT NULL,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT fk_social_user
        FOREIGN KEY (user_id)
        REFERENCES users(user_id),
    CONSTRAINT chk_social_provider
        CHECK (provider IN ('google', 'kakao'))
);

-- provider + provider_user_id = UNIQUE
CREATE UNIQUE INDEX idx_social_provider_user
ON social_account(provider, provider_user_id);

-- 4) DEVICE_TOKEN (푸시 알림 토큰 저장)

    CREATE TABLE device_token (
    token_id BIGSERIAL PRIMARY KEY,
    user_id BIGINT NOT NULL,
    fcm_token VARCHAR(500) NOT NULL,
    last_active_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT fk_device_token_user
        FOREIGN KEY (user_id)
        REFERENCES users(user_id)
);

-- 같은 user가 같은 token을 중복 등록하지 않도록
CREATE UNIQUE INDEX idx_device_user_token
ON device_token(user_id, fcm_token);

-- ============================================
-- 2. 다이어리 관련 테이블
-- ============================================

-- diary 테이블
CREATE TABLE diary (
    diary_id BIGSERIAL PRIMARY KEY UNIQUE,
    user_id BIGINT NOT NULL,
    title VARCHAR(100),
    content TEXT,
    summary_one_line VARCHAR(255),
    main_emotion VARCHAR(50) CHECK (main_emotion IN ('행복', '분노', '슬픔', '기쁨', '짜증','무난함')),
    emotion_score INT CHECK (emotion_score BETWEEN 0 AND 100),
    is_public BOOLEAN NOT NULL DEFAULT FALSE,
    conversation_duration_seconds INT,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    deleted_at TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE CASCADE
);

-- diary 테이블 인덱스
CREATE INDEX idx_diary_user_id ON diary(user_id, created_at);

-- diary_conversation 테이블
CREATE TABLE diary_conversation (
    msg_id BIGSERIAL PRIMARY KEY,
    diary_id BIGINT NOT NULL,
    sender VARCHAR(10) NOT NULL CHECK (sender IN ('USER', 'AI')),
    message_text TEXT NOT NULL,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (diary_id) REFERENCES diary(diary_id) ON DELETE CASCADE
);

-- sentence_analysis 테이블
CREATE TABLE sentence_analysis (
    analysis_id BIGSERIAL PRIMARY KEY,
    msg_id BIGINT NOT NULL,
    sentiment_label VARCHAR(50),
    confidence FLOAT CHECK (confidence BETWEEN 0.0 AND 1.0),
    sentence_text TEXT NOT NULL,
    sentence_index INT NOT NULL,
    FOREIGN KEY (msg_id) REFERENCES diary_conversation(msg_id) ON DELETE CASCADE
);

-- sentence_analysis 테이블 인덱스
CREATE INDEX idx_sentence_analysis_msg_id ON sentence_analysis(msg_id, sentence_index);

-- diary_attachment 테이블
CREATE TABLE diary_attachment (
    file_id BIGSERIAL PRIMARY KEY,
    diary_id BIGINT NOT NULL,
    file_url VARCHAR(500) NOT NULL,
    file_type VARCHAR(20) NOT NULL CHECK (file_type IN ('IMAGE', 'VOICE')),
    FOREIGN KEY (diary_id) REFERENCES diary(diary_id) ON DELETE CASCADE
);

-- diary_tag 테이블
CREATE TABLE diary_tag (
    tag_id BIGSERIAL PRIMARY KEY,
    diary_id BIGINT NOT NULL,
    tag_name VARCHAR(100) NOT NULL,
    tag_category VARCHAR(20) NOT NULL CHECK (tag_category IN ('ACTIVITY', 'PERSON', 'PLACE', 'TOPIC')),
    source_type VARCHAR(10) NOT NULL CHECK (source_type IN ('USER', 'AI')),
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (diary_id) REFERENCES diary(diary_id) ON DELETE CASCADE,
    UNIQUE (diary_id, tag_name, source_type)
);

-- ============================================
-- 3. 커뮤니티 관련 테이블
-- ============================================

-- post 테이블
CREATE TABLE post (
    post_id BIGSERIAL PRIMARY KEY,
    user_id BIGINT NOT NULL,
    diary_id BIGINT UNIQUE NOT NULL,
    post_title VARCHAR(100) NOT NULL,
    post_topic VARCHAR(100),
    post_emotion VARCHAR(50) CHECK (post_emotion IN ('행복', '분노', '슬픔', '기쁨', '짜증','무난함')),
    post_content TEXT NOT NULL,
    like_count INT NOT NULL DEFAULT 0,
    comment_count INT NOT NULL DEFAULT 0,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP,
    deleted_at TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE CASCADE,
    FOREIGN KEY (diary_id) REFERENCES diary(diary_id) ON DELETE CASCADE
);

-- post_like 테이블
CREATE TABLE post_like (
    like_id BIGSERIAL PRIMARY KEY,
    post_id BIGINT NOT NULL,
    user_id BIGINT NOT NULL,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (post_id) REFERENCES post(post_id) ON DELETE CASCADE,
    FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE CASCADE,
    UNIQUE (post_id, user_id)
);

-- post_comment 테이블
CREATE TABLE post_comment (
    comment_id BIGSERIAL PRIMARY KEY,
    post_id BIGINT NOT NULL,
    user_id BIGINT NOT NULL,
    comment_text TEXT NOT NULL,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP,
    deleted_at TIMESTAMP,
    FOREIGN KEY (post_id) REFERENCES post(post_id) ON DELETE CASCADE,
    FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE CASCADE
);

-- post_comment 테이블 인덱스
CREATE INDEX idx_post_comment_post_id ON post_comment(post_id, created_at);

-- post_attachment 테이블
CREATE TABLE post_attachment (
    file_id BIGSERIAL PRIMARY KEY,
    post_id BIGINT NOT NULL,
    file_url VARCHAR(500) NOT NULL,
    FOREIGN KEY (post_id) REFERENCES post(post_id) ON DELETE CASCADE
);

-- ============================================
-- 4. 통계 및 기타 (알림, 내보내기) 테이블
-- ============================================

-- monthly_report 테이블
CREATE TABLE monthly_report (
    report_id BIGSERIAL PRIMARY KEY,
    user_id BIGINT NOT NULL,
    report_year_month CHAR(7) NOT NULL,
    analysis_json JSONB NOT NULL,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE CASCADE,
    UNIQUE (user_id, report_year_month)
);

-- export_log 테이블
CREATE TABLE export_log (
    log_id BIGSERIAL PRIMARY KEY,
    user_id BIGINT NOT NULL,
    diary_id BIGINT,
    status VARCHAR(20) NOT NULL CHECK (status IN ('SUCCESS', 'FAILED', 'PENDING')),
    export_file_url VARCHAR(500),
    requested_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE CASCADE,
    FOREIGN KEY (diary_id) REFERENCES diary(diary_id) ON DELETE SET NULL
);

-- notification_log 테이블
CREATE TABLE notification_log (
    log_id BIGSERIAL PRIMARY KEY,
    user_id BIGINT NOT NULL,
    sent_date DATE NOT NULL,
    notification_type VARCHAR(50) NOT NULL,
    is_success BOOLEAN NOT NULL,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE CASCADE,
    UNIQUE (user_id, notification_type, sent_date)
);

-- ============================================
-- 인덱스 추가 (성능 최적화)
-- ============================================


-- diary 테이블 추가 인덱스
CREATE INDEX idx_diary_deleted_at ON diary(deleted_at);
CREATE INDEX idx_diary_created_at ON diary(created_at);

-- post 테이블 인덱스
CREATE INDEX idx_post_user_id ON post(user_id);
CREATE INDEX idx_post_created_at ON post(created_at);

-- post_like 테이블 인덱스
CREATE INDEX idx_post_like_user_id ON post_like(user_id);

-- post_comment 테이블 추가 인덱스
CREATE INDEX idx_post_comment_user_id ON post_comment(user_id);
